## 3. 核心算法设计

### 3.1 关系路径查找算法

使用广度优先搜索（BFS）查找两个人物之间的关系路径。

```javascript
/**
 * 查找关系路径
 * @param {String} fromId - 起始人物ID
 * @param {String} toId - 目标人物ID
 * @param {Number} maxDepth - 最大搜索深度
 * @return {Array} 路径列表
 */
async function findPath(fromId, toId, maxDepth = 6) {
  // 1. 初始化
  const queue = [{ characterId: fromId, path: [], depth: 0 }];
  const visited = new Set([fromId]);
  const allPaths = [];

  // 2. BFS搜索
  while (queue.length > 0) {
    const { characterId, path, depth } = queue.shift();

    // 深度限制
    if (depth >= maxDepth) continue;

    // 查询该人物的所有关系
    const guanxiList = await db.collection('guanxi')
      .where({
        $or: [
          { fromCharacterId: characterId },
          { toCharacterId: characterId }
        ],
        isActive: true
      })
      .get();

    // 3. 遍历关系
    for (const guanxi of guanxiList.data) {
      // 确定另一端的人物
      const nextCharacterId = guanxi.fromCharacterId === characterId
        ? guanxi.toCharacterId
        : guanxi.fromCharacterId;

      // 构建路径节点
      const pathNode = {
        characterId: nextCharacterId,
        characterName: guanxi.fromCharacterId === characterId
          ? guanxi.toCharacterName
          : guanxi.fromCharacterName,
        guanxiId: guanxi._id,
        guanxiType: guanxi.typeName,
        direction: guanxi.fromCharacterId === characterId ? 'to' : 'from'
      };

      const newPath = [...path, pathNode];

      // 4. 找到目标
      if (nextCharacterId === toId) {
        allPaths.push(newPath);
        continue;  // 继续查找其他路径
      }

      // 5. 未访问过则加入队列
      if (!visited.has(nextCharacterId)) {
        visited.add(nextCharacterId);
        queue.push({
          characterId: nextCharacterId,
          path: newPath,
          depth: depth + 1
        });
      }
    }
  }

  // 6. 排序（按路径长度）
  allPaths.sort((a, b) => a.length - b.length);

  return {
    paths: allPaths,
    shortestPath: allPaths[0] || null,
    pathCount: allPaths.length
  };
}
```

### 3.2 关系图谱生成算法

```javascript
/**
 * 生成关系图谱数据
 * @param {String} centerCharacterId - 中心人物ID
 * @param {Object} filters - 过滤条件
 * @return {Object} 图谱数据（nodes + edges）
 */
async function generateGraph(centerCharacterId, filters = {}) {
  const {
    typeIds = [],
    timePoint = null,
    maxDepth = 3,
    maxNodes = 200
  } = filters;

  // 1. 初始化
  const nodes = new Map();  // characterId -> node
  const edges = [];
  const queue = [{ characterId: centerCharacterId, depth: 0 }];
  const visited = new Set([centerCharacterId]);

  // 2. 添加中心节点
  const centerCharacter = await getCharacter(centerCharacterId);
  nodes.set(centerCharacterId, {
    id: centerCharacterId,
    name: centerCharacter.name,
    avatar: centerCharacter.avatar,
    isCenter: true,
    depth: 0
  });

  // 3. BFS扩展
  while (queue.length > 0 && nodes.size < maxNodes) {
    const { characterId, depth } = queue.shift();

    if (depth >= maxDepth) continue;

    // 查询关系
    const queryCondition = {
      $or: [
        { fromCharacterId: characterId },
        { toCharacterId: characterId }
      ]
    };

    // 类型筛选
    if (typeIds.length > 0) {
      queryCondition.typeId = { $in: typeIds };
    }

    const guanxiList = await db.collection('guanxi')
      .where(queryCondition)
      .get();

    // 4. 处理关系
    for (const guanxi of guanxiList.data) {
      // 时间点筛选
      if (timePoint && !isActiveAt(guanxi, timePoint)) {
        continue;
      }

      // 确定另一端人物
      const isFrom = guanxi.fromCharacterId === characterId;
      const nextCharacterId = isFrom
        ? guanxi.toCharacterId
        : guanxi.fromCharacterId;
      const nextCharacterName = isFrom
        ? guanxi.toCharacterName
        : guanxi.fromCharacterName;
      const nextCharacterAvatar = isFrom
        ? guanxi.toCharacterAvatar
        : guanxi.fromCharacterAvatar;

      // 添加节点
      if (!nodes.has(nextCharacterId)) {
        nodes.set(nextCharacterId, {
          id: nextCharacterId,
          name: nextCharacterName,
          avatar: nextCharacterAvatar,
          depth: depth + 1
        });
      }

      // 添加边
      edges.push({
        id: guanxi._id,
        source: isFrom ? characterId : nextCharacterId,
        target: isFrom ? nextCharacterId : characterId,
        type: guanxi.typeId,
        typeName: guanxi.typeName,
        label: guanxi.attributes?.title || guanxi.typeName,
        strength: guanxi.strength
      });

      // 加入队列
      if (!visited.has(nextCharacterId) && nodes.size < maxNodes) {
        visited.add(nextCharacterId);
        queue.push({
          characterId: nextCharacterId,
          depth: depth + 1
        });
      }
    }
  }

  // 5. 返回结果
  return {
    nodes: Array.from(nodes.values()),
    edges: edges,
    stats: {
      nodeCount: nodes.size,
      edgeCount: edges.length,
      maxDepthReached: Math.max(...Array.from(nodes.values()).map(n => n.depth))
    }
  };
}

/**
 * 判断关系在指定时间点是否活跃
 */
function isActiveAt(guanxi, timePoint) {
  const targetTime = new Date(timePoint);
  return guanxi.periods.some(period => {
    const start = new Date(period.startTime);
    const end = period.endTime ? new Date(period.endTime) : new Date();
    return start <= targetTime && targetTime <= end;
  });
}
```

### 3.3 时间段重叠检测算法

```javascript
/**
 * 检测时间段是否重叠
 * @param {Array} periods - 现有时间段列表
 * @param {Object} newPeriod - 新时间段
 * @return {Boolean} 是否重叠
 */
function hasOverlap(periods, newPeriod) {
  const newStart = new Date(newPeriod.startTime);
  const newEnd = newPeriod.endTime ? new Date(newPeriod.endTime) : new Date('9999-12-31');

  return periods.some(period => {
    const start = new Date(period.startTime);
    const end = period.endTime ? new Date(period.endTime) : new Date('9999-12-31');

    // 检测重叠：
    // 新时间段的开始在现有时间段内 或
    // 新时间段的结束在现有时间段内 或
    // 新时间段完全包含现有时间段
    return (
      (newStart >= start && newStart <= end) ||
      (newEnd >= start && newEnd <= end) ||
      (newStart <= start && newEnd >= end)
    );
  });
}

/**
 * 按时间排序并插入新时间段
 */
function insertPeriod(periods, newPeriod) {
  periods.push(newPeriod);
  periods.sort((a, b) => {
    return new Date(a.startTime) - new Date(b.startTime);
  });
  return periods;
}
```

### 3.4 关系强度计算算法

```javascript
/**
 * 计算关系强度（用于图谱可视化）
 * @param {Object} guanxi - 关系对象
 * @return {Number} 强度值 (0-1)
 */
function calculateGuanxiStrength(guanxi) {
  let score = 0;

  // 1. 基础强度
  const strengthMap = { strong: 0.5, medium: 0.3, weak: 0.1 };
  score += strengthMap[guanxi.strength] || 0.3;

  // 2. 亲密度
  if (guanxi.intimacy) {
    score += guanxi.intimacy / 200;  // 0-100 -> 0-0.5
  }

  // 3. 联系频率
  const frequencyMap = { often: 0.2, sometimes: 0.1, rarely: 0 };
  score += frequencyMap[guanxi.contactFrequency] || 0;

  // 4. 最后联系时间（越近越强）
  if (guanxi.lastContactTime) {
    const daysSince = (Date.now() - new Date(guanxi.lastContactTime)) / (1000 * 60 * 60 * 24);
    if (daysSince < 7) score += 0.1;
    else if (daysSince < 30) score += 0.05;
  }

  // 5. 时间段数量（关系越复杂越重要）
  score += Math.min(guanxi.periods.length * 0.05, 0.2);

  return Math.min(score, 1);  // 最大为1
}
```

### 3.5 生日属性推算算法

#### 3.5.1 阴阳历转换算法

使用第三方农历库（如 `lunar-javascript` 或 `solarlunar`）进行阴阳历转换。

```javascript
/**
 * 阳历转阴历
 * @param {String} solarDate - 阳历日期 (YYYY-MM-DD)
 * @return {String} 阴历日期 (YYYY-MM-DD)
 */
function solarToLunar(solarDate) {
  const Lunar = require('lunar-javascript');
  const solar = Lunar.Solar.fromYmd(
    parseInt(solarDate.substring(0, 4)),
    parseInt(solarDate.substring(5, 7)),
    parseInt(solarDate.substring(8, 10))
  );
  const lunar = solar.getLunar();

  return `${lunar.getYear()}-${String(lunar.getMonth()).padStart(2, '0')}-${String(lunar.getDay()).padStart(2, '0')}`;
}

/**
 * 阴历转阳历
 * @param {String} lunarDate - 阴历日期 (YYYY-MM-DD)
 * @return {String} 阳历日期 (YYYY-MM-DD)
 */
function lunarToSolar(lunarDate) {
  const Lunar = require('lunar-javascript');
  const lunar = Lunar.Lunar.fromYmd(
    parseInt(lunarDate.substring(0, 4)),
    parseInt(lunarDate.substring(5, 7)),
    parseInt(lunarDate.substring(8, 10))
  );
  const solar = lunar.getSolar();

  return `${solar.getYear()}-${String(solar.getMonth()).padStart(2, '0')}-${String(solar.getDay()).padStart(2, '0')}`;
}

/**
 * 处理生日输入并自动转换
 * @param {Object} input - 用户输入 { value, type }
 * @return {Object} 完整生日对象
 */
function processBirthday(input) {
  const { value, type } = input; // type: 'solar' | 'lunar'

  const result = {
    solar: type === 'solar' ? value : lunarToSolar(value),
    lunar: type === 'lunar' ? value : solarToLunar(value),
    primary: type
  };

  return result;
}
```

#### 3.5.2 星座计算算法

根据阳历生日计算星座。

```javascript
/**
 * 根据阳历生日计算星座
 * @param {String} solarDate - 阳历日期 (YYYY-MM-DD)
 * @return {String} 星座英文名
 */
function calculateZodiacSign(solarDate) {
  const month = parseInt(solarDate.substring(5, 7));
  const day = parseInt(solarDate.substring(8, 10));

  const zodiacSigns = [
    { name: 'Capricorn', start: [12, 22], end: [1, 19] },
    { name: 'Aquarius', start: [1, 20], end: [2, 18] },
    { name: 'Pisces', start: [2, 19], end: [3, 20] },
    { name: 'Aries', start: [3, 21], end: [4, 19] },
    { name: 'Taurus', start: [4, 20], end: [5, 20] },
    { name: 'Gemini', start: [5, 21], end: [6, 21] },
    { name: 'Cancer', start: [6, 22], end: [7, 22] },
    { name: 'Leo', start: [7, 23], end: [8, 22] },
    { name: 'Virgo', start: [8, 23], end: [9, 22] },
    { name: 'Libra', start: [9, 23], end: [10, 23] },
    { name: 'Scorpio', start: [10, 24], end: [11, 22] },
    { name: 'Sagittarius', start: [11, 23], end: [12, 21] }
  ];

  for (const sign of zodiacSigns) {
    const [startMonth, startDay] = sign.start;
    const [endMonth, endDay] = sign.end;

    // 处理跨年的情况（摩羯座）
    if (startMonth > endMonth) {
      if ((month === startMonth && day >= startDay) ||
          (month === endMonth && day <= endDay)) {
        return sign.name;
      }
    } else {
      if ((month === startMonth && day >= startDay) ||
          (month === endMonth && day <= endDay) ||
          (month > startMonth && month < endMonth)) {
        return sign.name;
      }
    }
  }

  return 'Unknown';
}
```

#### 3.5.3 生肖计算算法

根据农历年份计算生肖。

```javascript
/**
 * 根据农历年份计算生肖
 * @param {String} lunarDate - 阴历日期 (YYYY-MM-DD)
 * @return {String} 生肖英文名
 */
function calculateChineseZodiac(lunarDate) {
  const year = parseInt(lunarDate.substring(0, 4));

  // 生肖轮回周期为12年，起始年份1900年为鼠年
  const zodiacAnimals = [
    'Rat', 'Ox', 'Tiger', 'Rabbit',
    'Dragon', 'Snake', 'Horse', 'Goat',
    'Monkey', 'Rooster', 'Dog', 'Pig'
  ];

  // 计算相对于1900年的偏移
  const offset = (year - 1900) % 12;
  return zodiacAnimals[offset];
}
```

#### 3.5.4 完整的生日处理流程

用户创建人物时的完整处理逻辑：

```javascript
/**
 * 处理人物生日及派生属性
 * @param {Object} birthdayInput - 生日输入
 * @return {Object} 完整的生日和派生属性
 */
async function processBirthdayAttributes(birthdayInput) {
  // 1. 转换阴阳历
  const birthday = processBirthday(birthdayInput);

  // 2. 计算星座（基于阳历）
  const zodiacSign = calculateZodiacSign(birthday.solar);

  // 3. 计算生肖（基于阴历）
  const chineseZodiac = calculateChineseZodiac(birthday.lunar);

  // 4. 返回完整数据
  return {
    birthday: birthday,
    immutableAttributes: {
      zodiacSign: zodiacSign,
      chineseZodiac: chineseZodiac
      // birthPlace, birthNationality, ethnicity, bloodType 由用户输入
    }
  };
}

/**
 * 验证生日合法性
 * @param {String} date - 日期字符串
 * @param {String} type - 类型：solar/lunar
 * @return {Object} 验证结果
 */
function validateBirthday(date, type) {
  // 1. 日期格式验证
  const dateRegex = /^\d{4}-\d{2}-\d{2}$/;
  if (!dateRegex.test(date)) {
    return { valid: false, error: '日期格式错误，应为 YYYY-MM-DD' };
  }

  // 2. 日期范围验证
  const year = parseInt(date.substring(0, 4));
  if (year < 1900 || year > 2100) {
    return { valid: false, error: '年份必须在 1900-2100 之间' };
  }

  // 3. 阳历日期合法性验证
  if (type === 'solar') {
    const dateObj = new Date(date);
    if (isNaN(dateObj.getTime())) {
      return { valid: false, error: '阳历日期不合法' };
    }
  }

  // 4. 阴历日期合法性验证
  if (type === 'lunar') {
    try {
      lunarToSolar(date);  // 尝试转换，如果失败说明不合法
    } catch (e) {
      return { valid: false, error: '阴历日期不合法' };
    }
  }

  return { valid: true };
}
```
