#### 7.3.2 初始化本地数据

小程序首次启动时会自动执行以下初始化：

```javascript
// miniprogram/app.js
App({
  async onLaunch() {
    try {
      // 1. 初始化 IndexedDB
      const storage = require('./services/storage');
      await storage.init();
      console.log('[App] IndexedDB 初始化成功');
      
      // 2. 预装关系类型到本地数据库
      const { preloadTypes } = require('./models/guanxiTypes');
      await preloadTypes();
      console.log('[App] 关系类型预装成功');
      
      // 3. 检查数据库版本并执行迁移
      await this.checkDatabaseVersion();
      
      // 4. 初始化用户信息
      await this.initUserInfo();
      
      console.log('[App] 应用初始化完成');
    } catch (error) {
      console.error('[App] 初始化失败:', error);
      wx.showToast({
        title: '应用初始化失败',
        icon: 'none'
      });
    }
  },
  
  async checkDatabaseVersion() {
    const currentVersion = 2; // 当前数据库版本
    const storage = require('./services/storage');
    const storedVersion = await storage.getItem('db_version');
    
    if (storedVersion < currentVersion) {
      console.log(`[App] 数据库需要升级: ${storedVersion} -> ${currentVersion}`);
      await this.migrateDatabase(storedVersion, currentVersion);
      await storage.setItem('db_version', currentVersion);
    }
  },
  
  async initUserInfo() {
    // 获取或创建用户信息
    const userService = require('./services/userService');
    let user = await userService.getCurrentUser();
    
    if (!user) {
      user = await userService.createDefaultUser();
      console.log('[App] 创建默认用户');
    }
    
    this.globalData.user = user;
  }
});
```

**预装的关系类型**：
- `family_relative` - 亲属关系
- `social_friend` - 好友关系
- `work_colleague` - 同事关系
- `education_classmate` - 同学关系
- `neighbor_community` - 邻里关系

#### 7.3.5 提交审核

1. **登录微信公众平台**
   - 访问 https://mp.weixin.qq.com/
   - 进入"版本管理"页面

2. **选择开发版本提交审核**
   - 选择刚上传的版本
   - 点击"提交审核"
   - 填写审核信息

3. **配置审核信息**
   ```
   - 服务类目: 工具 > 效率工具
   - 标签: 人际关系、关系管理、人脉管理
   - 功能页面截图: 准备 5 张核心功能截图
   - 核心功能描述: 详细说明关系管理功能
   - 隐私政策: 提供隐私政策链接
   ```

4. **审核注意事项**
   - 确保功能完整可用

   - 说明云同步为可选功能

```javascript
// scripts/init-cloud-database.js
const cloud = require('wx-server-sdk');

cloud.init({
  env: 'your-cloud-env-id'

const db = cloud.database();

async function initCollections() {
  const collections = [
    'users',

    'character_temporal_attributes',
    'guanxi_types',

    'reminders',
    'events',
    'graph_snapshots'

      // 检查集合是否存在

      console.log(`集合 ${name} 已存在`);
```

#### 7.4.5 创建云数据库索引

```javascript
// scripts/create-indexes.js
const cloud = require('wx-server-sdk');
cloud.init({ env: 'your-cloud-env-id' });
const db = cloud.database();

async function createIndexes() {
  // characters 集合索引
  await db.collection('characters').createIndex({
    keys: { userId: 1 },
    name: 'userId_index'
  });
  
  await db.collection('characters').createIndex({
    keys: { userId: 1, name: 1 },
    name: 'userId_name_index'
  });
  
  await db.collection('characters').createIndex({
    keys: { userId: 1, updatedAt: -1 },
    name: 'userId_updatedAt_index'
  });
  
  // guanxi 集合索引
  await db.collection('guanxi').createIndex({
    keys: { userId: 1, fromCharacterId: 1 },
    name: 'userId_from_index'
  });
  
  await db.collection('guanxi').createIndex({
    keys: { userId: 1, toCharacterId: 1 },
    name: 'userId_to_index'
  });
  
  await db.collection('guanxi').createIndex({
    keys: { userId: 1, 'periods.startTime': -1 },
    name: 'userId_startTime_index'
  });
  
  // reminders 集合索引
  await db.collection('reminders').createIndex({
    keys: { userId: 1, triggerTime: 1 },
    name: 'userId_triggerTime_index'
  });
  
  console.log('所有索引创建完成');
}

createIndexes().catch(console.error);
```

### 7.6 版本更新流程

#### 7.6.1 数据库结构变更

当 IndexedDB 结构发生变化时：

```javascript
// miniprogram/services/storage.js
class IndexedDBService {
  constructor() {
    this.dbName = 'guanxi_db';
    this.version = 3;  // ⚠️ 增加版本号
  }
  
  async init() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(this.dbName, this.version);
      
      request.onerror = () => reject(request.error);
      request.onsuccess = () => {
        this.db = request.result;
        resolve(this.db);
      };
      
      // 处理版本升级
      request.onupgradeneeded = (event) => {
        const db = event.target.result;
        const oldVersion = event.oldVersion;
        const newVersion = event.newVersion;
        
        console.log(`数据库升级: ${oldVersion} -> ${newVersion}`);
        this.handleUpgrade(db, oldVersion, newVersion);

    console.log(`执行数据库升级逻辑...`);
    
    // 版本 1 -> 2: 添加 graph_snapshots 表
    if (oldVersion < 2) {
      const store = db.createObjectStore('graph_snapshots', {
        keyPath: 'id',
        autoIncrement: true
      });
      store.createIndex('userId', 'userId', { unique: false });
      store.createIndex('centerId', 'centerCharacterId', { unique: false });
      console.log('已添加 graph_snapshots 对象存储');
    }
    
    // 版本 2 -> 3: 为 characters 添加新索引
    if (oldVersion < 3) {
      const transaction = db.transaction;
      const store = transaction.objectStore('characters');
      
      if (!store.indexNames.contains('culturalContext')) {
        store.createIndex('culturalContext', 'culturalContext', { unique: false });
        console.log('已为 characters 添加 culturalContext 索引');
      }
    }
  }
  
  async migrateData(oldVersion, newVersion) {
    // 数据迁移逻辑
    if (oldVersion < 2 && newVersion >= 2) {
      await this.migrateToV2();
    }
    if (oldVersion < 3 && newVersion >= 3) {
      await this.migrateToV3();
    }
  }
  
  async migrateToV2() {
    console.log('执行 V2 数据迁移...');
    // 迁移逻辑
  }
  
  async migrateToV3() {
    console.log('执行 V3 数据迁移...');
    // 为所有现有人物添加默认 culturalContext
    const characters = await this.getAllCharacters();
    for (const char of characters) {
      if (!char.culturalContext) {
        char.culturalContext = 'zh-CN';
```

```javascript
// services/logger.js
class Logger {
  constructor() {
    this.logLevel = {
      debug: 0,
      info: 1,
      warn: 2,
      error: 3
    };
    
    this.currentLevel = this.getLogLevel();
    this.maxLogs = 1000;  // 最大日志条数
    this.logs = [];
  }
  
  getLogLevel() {
    const config = require('../config');
    return this.logLevel[config.logLevel] || this.logLevel.info;
  }
  
  log(level, message, data = {}) {
    if (this.logLevel[level] < this.currentLevel) {
      return;
    }
    
    const logEntry = {
      timestamp: new Date().toISOString(),
      level,
      message,
      data,
      page: getCurrentPages().pop()?.route || 'unknown'
    };
    
    // 输出到控制台
    console[level](` [${level.toUpperCase()}]`, message, data);
    
    // 保存到内存
    this.logs.push(logEntry);
    if (this.logs.length > this.maxLogs) {
      this.logs.shift();
    }
    
    // 错误级别立即上报
    if (level === 'error') {
      this.reportError(logEntry);
    }
  }
  
  debug(message, data) {
    this.log('debug', message, data);
  }
  
  info(message, data) {
    this.log('info', message, data);
  }
  
  warn(message, data) {
    this.log('warn', message, data);
  }
  
  error(message, data) {
    this.log('error', message, data);
  }
  
  async reportError(logEntry) {
    try {
      // 上报到云函数
      if (wx.cloud) {
        await wx.cloud.callFunction({
          name: 'reportError',
          data: logEntry
        });
      }
    } catch (err) {
      console.error('日志上报失败:', err);
    }
  }
  
  exportLogs() {
    return JSON.stringify(this.logs, null, 2);
  }
  
  clearLogs() {
    this.logs = [];
  }
}

module.exports = new Logger();
```

```javascript
class PerformanceMonitor {
  constructor() {
    this.metrics = {};
  }
  
  start(label) {
    this.metrics[label] = {
      startTime: Date.now(),
      startMemory: wx.getPerformance?.()?.memory?.usedJSHeapSize || 0
    };
  }
  
  end(label) {
    if (!this.metrics[label]) {
      console.warn(`性能指标 ${label} 未开始`);
      return;
    }
    
    const metric = this.metrics[label];
    const endTime = Date.now();
    const endMemory = wx.getPerformance?.()?.memory?.usedJSHeapSize || 0;
    
    const result = {
      label,
      duration: endTime - metric.startTime,
      memoryDelta: endMemory - metric.startMemory
    };
    
    console.log(`[Performance] ${label}:`, result);
    
    // 性能异常上报
    if (result.duration > 3000) {  // 超过3秒
      this.reportSlowOperation(result);
    }
    
    delete this.metrics[label];
    return result;
  }
  
  async reportSlowOperation(result) {
    const logger = require('../services/logger');
    logger.warn('操作性能异常', result);
  }
  
  // 监控页面性能
  monitorPagePerformance(pageName) {
    const performance = wx.getPerformance?.();
    if (!performance) return;
    
    const observer = performance.createObserver((entryList) => {
      const entries = entryList.getEntries();
      entries.forEach((entry) => {
        console.log(`[PagePerformance] ${pageName}:`, {
          name: entry.name,
          duration: entry.duration,
          startTime: entry.startTime
        });
      });
    });
    
    observer.observe({ entryTypes: ['render', 'script', 'navigation'] });
  }
}

module.exports = new PerformanceMonitor();
```

#### 7.9.1 IndexedDB 优化

```javascript
// 使用事务批量操作
async function batchCreateCharacters(characters) {
  return new Promise((resolve, reject) => {
    const transaction = db.transaction(['characters'], 'readwrite');
    const store = transaction.objectStore('characters');
    
    const results = [];
    characters.forEach(char => {
      const request = store.add(char);
      request.onsuccess = () => results.push(request.result);
    });
    
    transaction.oncomplete = () => resolve(results);
    transaction.onerror = () => reject(transaction.error);
  });
}

// 使用索引优化查询
async function queryByUserId(userId) {
  const store = db.transaction('characters', 'readonly').objectStore('characters');
  const index = store.index('userId');
  const request = index.getAll(userId);
  
  return new Promise((resolve, reject) => {
    request.onsuccess = () => resolve(request.result);
    request.onerror = () => reject(request.error);
  });
}

// 使用游标分页查询
async function queryWithPagination(userId, pageSize, cursor) {
  const store = db.transaction('characters', 'readonly').objectStore('characters');
  const index = store.index('userId');
  const range = IDBKeyRange.only(userId);
  const request = index.openCursor(range, 'next');
  
  return new Promise((resolve, reject) => {
    const results = [];
    let count = 0;
    let skipped = 0;
    
    request.onsuccess = (event) => {
      const cursor = event.target.result;
      
      if (!cursor) {
        resolve({ data: results, hasMore: count === pageSize });
        return;
      }
      
      if (skipped < cursor) {
        skipped++;
        cursor.continue();
        return;
      }
      
      if (count < pageSize) {
        results.push(cursor.value);
        count++;
        cursor.continue();
      } else {
        resolve({ data: results, hasMore: true });
      }
    };
    
    request.onerror = () => reject(request.error);
  });
}
```

```javascript
    const cacheKey = this.getCacheKey(centerCharacterId, filters);
    
    // 尝试从缓存获取
    if (useCache) {

    return { centerId, directGuanxi };
  }
  
  // 限制节点数量
  async loadStage2(stage1, filters) {
    const maxNodes = filters.maxNodes || 100;
    // 根据关系强度排序，只取前 maxNodes 个
    const sortedGuanxi = stage1.directGuanxi
      .sort((a, b) => b.strength - a.strength)
      .slice(0, maxNodes);
      
    return { ...stage1, guanxi: sortedGuanxi };
  }
}
```

```javascript
// services/validatorService.js
class ValidatorService {
  validateCharacterInput(data) {
    const errors = [];
    
    // 名称验证
    if (!data.name || data.name.trim().length === 0) {
      errors.push('名称不能为空');
    }
    if (data.name.length > 50) {
      errors.push('名称不能超过50个字符');
    }
    
    // XSS 防护
    data.name = this.escapeHTML(data.name);
    data.note = this.escapeHTML(data.note);
    
    return { valid: errors.length === 0, errors, data };
  }
  
  escapeHTML(str) {
    if (!str) return '';
    return str
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;')
      .replace(/'/g, '&#x27;');
  }
}

module.exports = new ValidatorService();
```

#### 7.11.1 自动备份策略

```javascript
// services/backupService.js
class BackupService {
  async autoBackup() {
    const config = require('../config');
    if (!config.enableCloudBackup) {
      return;
    }
    
    try {
      // 检查上次备份时间
      const lastBackup = await this.getLastBackupTime();
      const now = Date.now();
      const daysSinceBackup = (now - lastBackup) / (1000 * 60 * 60 * 24);
      
      // 每7天自动备份一次
      if (daysSinceBackup >= 7) {
        await this.createBackup();
      }
    } catch (error) {
      console.error('自动备份失败:', error);
    }
  }
  
  async createBackup() {
    const storage = require('./storage');
    
    // 导出所有数据
    const data = await storage.exportAll();
    
    // 上传到云端
    const result = await wx.cloud.callFunction({
      name: 'backup',
      data: {
        operation: 'create',
        data: data,
        timestamp: Date.now()
      }
    });
    
    if (result.result.success) {
      await this.saveBackupRecord(result.result.backupId);
    }
    
    return result;
  }
  
  async listBackups() {
    const result = await wx.cloud.callFunction({
      name: 'backup',
      data: { operation: 'list' }
    });
    
    return result.result.backups;
  }
  
  async restoreFromBackup(backupId) {
    // 下载备份数据
    const result = await wx.cloud.callFunction({
      name: 'backup',
      data: {
        operation: 'get',
        backupId
      }
    });
    
    if (!result.result.success) {
      throw new Error('备份恢复失败');
    }
    
    // 清空本地数据
    const storage = require('./storage');
    await storage.clearAll();
    
    // 导入备份数据
    await storage.importAll(result.result.data);
    
    return { success: true };
  }
}

module.exports = new BackupService();
```

```javascript
// utils/featureFlag.js
class FeatureFlag {
  constructor() {
    this.features = {};
  }
  
  async init() {
    // 从云端获取功能开关配置
    try {
      const result = await wx.cloud.callFunction({
        name: 'getFeatureFlags'
      });
      this.features = result.result.features;
    } catch (error) {
      console.error('获取功能开关失败:', error);
      this.features = this.getDefaultFeatures();
    }
  }
  
  isEnabled(feature, userId) {
    const config = this.features[feature];
    if (!config) return false;
    
    // 全局开关
    if (!config.enabled) return false;
    
    // 灰度比例
    if (config.grayPercent < 100) {
      const hash = this.hashUserId(userId);
      if (hash > config.grayPercent) {
        return false;
      }
    }
    
    // 白名单
    if (config.whitelist && config.whitelist.includes(userId)) {
      return true;
    }
    
    return config.enabled;
  }
  
  hashUserId(userId) {
    // 简单哈希，返回 0-100
    let hash = 0;
    for (let i = 0; i < userId.length; i++) {
      hash = ((hash << 5) - hash) + userId.charCodeAt(i);
      hash = hash & hash;
    }
    return Math.abs(hash) % 100;
  }
  
  getDefaultFeatures() {
    return {
      relationDeduction: { enabled: true, grayPercent: 100 },
      cloudSync: { enabled: true, grayPercent: 100 },
      advancedGraph: { enabled: false, grayPercent: 10 }
    };
  }
}

module.exports = new FeatureFlag();
```
