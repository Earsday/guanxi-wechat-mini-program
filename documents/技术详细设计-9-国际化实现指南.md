## 9. 国际化实现指南

本章详细说明如何实现"人际关系图谱"小程序的完整国际化支持，涵盖界面多语言、人物属性国际化、日期时间国际化、数据导出国际化等方面。

### 9.1 国际化架构概述

#### 9.1.1 多层次国际化设计

```
┌─────────────────────────────────────┐
│       界面层国际化（UI i18n）         │
│   - 界面文本翻译                      │
│   - 动态语言切换                      │
└─────────────────────────────────────┘
                 ↓
┌─────────────────────────────────────┐
│      数据层国际化（Data i18n）        │
│   - 人物属性本地化                    │
│   - 关系类型本地化                    │
│   - 文化背景适配                      │
└─────────────────────────────────────┐
                 ↓
┌─────────────────────────────────────┐
│    格式层国际化（Format i18n）        │
│   - 日期时间格式                      │
│   - 姓名显示格式                      │
│   - 地址格式                          │
└─────────────────────────────────────┘
```

### 9.3 人物属性国际化

#### 9.3.1 姓名字段的国际化处理

**数据结构**（已在数据库设计中定义）：
```javascript
{
  name: String,          // 完整姓名（向后兼容）
  displayName: String,   // 显示名称（优先使用）

    // 中文格式
    lastName: String,         // 姓
    firstName: String,        // 名
    
    // 西方格式
    firstNameWestern: String,
    middleNameWestern: String,
    lastNameWestern: String,
    
    // 日语格式
    lastNameJapanese: String,
    firstNameJapanese: String
  },
  
  culturalContext: String  // 文化背景：zh-CN, en-US, ja-JP 等
```

**姓名显示逻辑**：
```javascript
// utils/nameFormatter.js

/**
 * 根据文化背景格式化姓名显示
 * @param {Object} character - 人物对象
 * @param {String} displayLocale - 显示语言（可选，默认使用当前语言）
 * @returns {String} 格式化后的姓名
 */
export function formatName(character, displayLocale = null) {
  const locale = displayLocale || i18n.locale;
  const { nameFields, culturalContext, displayName, name } = character;
  
  // 1. 优先使用用户自定义的 displayName
  if (displayName) {
    return displayName;
  }
  
  // 2. 根据文化背景格式化
  if (!nameFields) {
    return name || '';  // 兼容旧数据
  }
  
  // 中文姓名格式：姓 + 名
  if (culturalContext && culturalContext.startsWith('zh')) {
    const lastName = nameFields.lastName || '';
    const firstName = nameFields.firstName || '';
    return `${lastName}${firstName}`;
  }
  
  // 西方姓名格式：名 + 中间名 + 姓
  if (culturalContext && culturalContext.startsWith('en')) {
    const parts = [
      nameFields.firstNameWestern,
      nameFields.middleNameWestern,
      nameFields.lastNameWestern
    ].filter(Boolean);
    return parts.join(' ');
  }
  
  // 日语姓名格式：姓 + 名
  if (culturalContext === 'ja-JP') {
    const lastName = nameFields.lastNameJapanese || '';
    const firstName = nameFields.firstNameJapanese || '';
    return `${lastName} ${firstName}`;
  }
  
  // 默认：尝试组合可用字段
  return nameFields.lastName && nameFields.firstName
    ? `${nameFields.lastName}${nameFields.firstName}`
    : name || '';
}

/**
 * 获取姓名编辑表单字段配置
 * @param {String} culturalContext - 文化背景
 * @returns {Array} 表单字段配置
 */
export function getNameFormFields(culturalContext) {
  const fields = [];
  
  // 中文姓名字段
  if (!culturalContext || culturalContext.startsWith('zh')) {
    fields.push(
      { key: 'nameFields.lastName', label: i18n.t('nameFields.lastName'), required: true },
      { key: 'nameFields.firstName', label: i18n.t('nameFields.firstName'), required: true }
    );
  }
  
  // 西方姓名字段
  if (culturalContext && culturalContext.startsWith('en')) {
    fields.push(
      { key: 'nameFields.firstNameWestern', label: i18n.t('nameFields.firstNameWestern'), required: true },
      { key: 'nameFields.middleNameWestern', label: i18n.t('nameFields.middleNameWestern'), required: false },
      { key: 'nameFields.lastNameWestern', label: i18n.t('nameFields.lastNameWestern'), required: true }
    );
  }
  
  // 日语姓名字段
  if (culturalContext === 'ja-JP') {
    fields.push(
      { key: 'nameFields.lastNameJapanese', label: i18n.t('nameFields.lastNameJapanese'), required: true },
      { key: 'nameFields.firstNameJapanese', label: i18n.t('nameFields.firstNameJapanese'), required: true }
    );
  }
  
  return fields;
}
```

**在人物编辑页面中使用**：
```xml
<!-- pages/characters/edit/edit.wxml -->
```

```javascript
// pages/characters/edit/edit.js
import { getNameFormFields, formatName } from '../../../utils/nameFormatter';

Page({
  data: {
    character: {},
    nameFormFields: [],
    culturalContextIndex: 0,
    culturalContextOptions: []
  },
  
  onLoad(options) {
    // 加载文化背景选项
    const culturalContextOptions = [
      { value: 'zh-CN', label: i18n.t('character.culturalContextOptions.zh-CN') },
      { value: 'en-US', label: i18n.t('character.culturalContextOptions.en-US') },
      { value: 'ja-JP', label: i18n.t('character.culturalContextOptions.ja-JP') }
    ];
    
    this.setData({
      culturalContextOptions
    });
    
    // 加载人物数据（如果是编辑）
    if (options.id) {
      this.loadCharacter(options.id);
    } else {
      // 新建：使用默认文化背景
      this.updateNameFormFields('zh-CN');
    }
  },
  
  onCultureChange(e) {
    const index = e.detail.value;
    const culturalContext = this.data.culturalContextOptions[index].value;
    
    this.setData({
      culturalContextIndex: index,
      'character.culturalContext': culturalContext
    });
    
    this.updateNameFormFields(culturalContext);
  },
  
  updateNameFormFields(culturalContext) {
    const nameFormFields = getNameFormFields(culturalContext);
    this.setData({ nameFormFields });
  },
  
  getFieldValue(key) {
    // 解析嵌套键，如 "nameFields.lastName"
    const keys = key.split('.');
    let value = this.data.character;
    for (const k of keys) {
      value = value?.[k];
    }
    return value || '';
  },
  
  onSubmit(e) {
    const formData = e.detail.value;
    
    // 构建人物对象
    const character = {
      ...this.data.character,
      culturalContext: this.data.culturalContextOptions[this.data.culturalContextIndex].value,
      nameFields: {},
      displayName: formData.displayName
    };
    
    // 提取 nameFields
    Object.keys(formData).forEach(key => {
      if (key.startsWith('nameFields.')) {
        const fieldName = key.replace('nameFields.', '');
        character.nameFields[fieldName] = formData[key];
      }
    });
    
    // 自动生成 displayName（如果用户未填写）
    if (!character.displayName) {
      character.displayName = formatName(character);
    }
    
    // 保存人物
    this.saveCharacter(character);
  }
});
```

#### 9.3.3 联系方式的国际化

**地址格式化**：
```javascript
// utils/addressFormatter.js

/**
 * 根据国家格式化地址
 * @param {Object} address - 地址对象
 * @param {String} locale - 显示语言
 * @returns {String} 格式化后的地址
 */
export function formatAddress(address, locale = null) {
  const { country, province, city, address: street, postalCode } = address;
  
  // 中国地址格式：国家 省 市 街道 邮编
  if (country === 'CN' || locale?.startsWith('zh')) {
    return [country, province, city, street, postalCode].filter(Boolean).join(' ');
  }
  
  // 美国地址格式：街道, 城市, 省/州 邮编, 国家
  if (country === 'US' || locale === 'en-US') {
    return [street, city, `${province} ${postalCode}`, country].filter(Boolean).join(', ');
  }
  
  // 日本地址格式：〒邮编 国家 省 市 街道
  if (country === 'JP' || locale === 'ja-JP') {
    const parts = [postalCode ? `〒${postalCode}` : '', country, province, city, street];
    return parts.filter(Boolean).join(' ');
  }
  
  // 默认格式
  return [street, city, province, postalCode, country].filter(Boolean).join(', ');
}
```

### 9.4 日期时间国际化

#### 9.4.1 日期格式化

```javascript
// utils/dateFormatter.js
import dayjs from 'dayjs';
import 'dayjs/locale/zh-cn';
import 'dayjs/locale/en';
import 'dayjs/locale/ja';

/**
 * 格式化日期
 * @param {String|Date} date - 日期
 * @param {String} format - 格式（可选）
 * @param {String} locale - 语言（可选）
 * @returns {String} 格式化后的日期
 */
export function formatDate(date, format = null, locale = null) {
  if (!date) return '';
  
  const loc = locale || i18n.locale;
  const fmt = format || getDefaultDateFormat(loc);
  
  // 设置 dayjs 语言
  const dayjsLocale = {
    'zh-CN': 'zh-cn',
    'zh-TW': 'zh-cn',
    'en-US': 'en',
    'ja-JP': 'ja'
  }[loc] || 'en';
  
  return dayjs(date).locale(dayjsLocale).format(fmt);
}

/**
 * 获取默认日期格式
 * @param {String} locale - 语言
 * @returns {String} 日期格式字符串
 */
function getDefaultDateFormat(locale) {
  const formats = {
    'zh-CN': 'YYYY年MM月DD日',
    'zh-TW': 'YYYY年MM月DD日',
    'en-US': 'MM/DD/YYYY',
    'ja-JP': 'YYYY年MM月DD日'
  };
  
  return formats[locale] || 'YYYY-MM-DD';
}

/**
 * 格式化相对时间
 * @param {String|Date} date - 日期
 * @param {String} locale - 语言（可选）
 * @returns {String} 相对时间（如：3天前）
 */
export function formatRelativeTime(date, locale = null) {
  if (!date) return '';
  
  const loc = locale || i18n.locale;
  const dayjsLocale = {
    'zh-CN': 'zh-cn',
    'en-US': 'en',
    'ja-JP': 'ja'
  }[loc] || 'en';
  
  return dayjs(date).locale(dayjsLocale).fromNow();
}
```

```javascript
export function formatBirthday(birthday, locale = null) {
```

#### 9.5.2 获取本地化的关系类型名称

```javascript
// services/typeService.js

/**
 * 获取关系类型的本地化名称
 * @param {Object} type - 关系类型对象
 * @param {String} locale - 语言（可选）
 * @param {Boolean} reverse - 是否使用反向名称（可选）
 * @returns {String} 本地化的名称
 */
export function getLocalizedTypeName(type, locale = null, reverse = false) {
  const loc = locale || i18n.locale;
  const key = `${type.i18nKey}.${reverse ? 'reverseName' : 'name'}`;
  
  // 尝试获取翻译
  const translation = i18n.t(key);
  
  // 如果翻译不存在，回退到默认名称
  if (translation === key) {
    return reverse ? (type.reverseName || type.name) : type.name;
  }
  
  return translation;
}

/**
 * 获取关系类型的本地化描述
 * @param {Object} type - 关系类型对象
 * @param {String} locale - 语言（可选）
 * @returns {String} 本地化的描述
 */
export function getLocalizedTypeDescription(type, locale = null) {
  const loc = locale || i18n.locale;
  const key = `${type.i18nKey}.description`;
  
  const translation = i18n.t(key);
  if (translation === key) {
    return type.description || '';
  }
  
  return translation;
}
```

### 9.6 数据导出的国际化

#### 9.6.1 导出文件命名规范

**命名格式**：
```
guanxi_[英文名]_[本地语言名]_[UTC时间].gx0
```

**示例**：
- 中文单语言：`guanxi_我的族谱_20260321T083045Z.gx0`
- 中英双语：`guanxi_hongloumeng_红楼梦_20260321T083045Z.gx0`
- 英文单语言：`guanxi_my_network_20260321T083045Z.gx0`

**实现代码**：
```javascript
// services/exportService.js

/**
 * 生成导出文件名
 * @param {String} localName - 本地语言名称（必填）
 * @param {String} englishName - 英文名称（可选）
 * @returns {String} 文件名
 */
export function generateExportFileName(localName, englishName = null) {
  // UTC时间戳
  const timestamp = new Date().toISOString().replace(/[-:.]/g, '').slice(0, 15) + 'Z';
  
  // 构建文件名
  const parts = ['guanxi'];
  
  if (englishName) {
    parts.push(englishName);
  }
  
  parts.push(localName);
  parts.push(timestamp);
  
  return `${parts.join('_')}.gx0`;
}
```

### 9.8 最佳实践

#### 9.8.1 翻译键命名规范

1. **使用分层结构**：`模块.子模块.具体项`
   ```javascript
   character.edit.title
   guanxi.types.family
   ```

2. **避免过长的键名**：
   ```javascript
   // ❌ 不好
   character.detail.personal.information.section.title
   
   // ✅ 好
   character.detail.personalInfo
   ```

3. **使用语义化的键名**：
   ```javascript
   // ❌ 不好
   btn1, txt2, label3
   
   // ✅ 好
   common.save, character.name, guanxi.startTime
   ```

#### 9.8.2 处理复杂文本和变量

**使用插值**：

#### 9.8.3 性能优化

1. **懒加载语言包**：
   ```javascript
   // 仅在切换语言时加载新的语言包
   async function loadLocale(locale) {
     const messages = await import(`./locales/${locale}.js`);
     i18n.setLocaleMessage(locale, messages.default);
   }
   ```

2. **缓存翻译结果**：
   ```javascript
   const translationCache = new Map();
   
   function cachedTranslate(key, locale) {
     const cacheKey = `${locale}:${key}`;
     if (translationCache.has(cacheKey)) {
       return translationCache.get(cacheKey);
     }
     
     const translation = i18n.t(key);
     translationCache.set(cacheKey, translation);
     return translation;
   }
   ```

3. **预加载常用翻译**：
   ```javascript
   // 在应用启动时预加载常用翻译
   const commonKeys = [
     'common.save',
   ```
