## 11. 安全实施指南

本章详细说明"人际关系图谱"小程序的安全架构、安全威胁分析、防护措施和最佳实践，确保用户数据和隐私的安全。

### 11.1 安全架构概述

#### 11.1.1 安全设计原则

```
┌────────────────────────────────────────┐
│        纵深防御（Defense in Depth）      │
├────────────────────────────────────────┤
│ 1. 数据加密层 - 静态数据加密            │
│ 2. 传输安全层 - HTTPS/TLS              │
│ 3. 访问控制层 - 身份验证和授权          │
│ 4. 应用安全层 - 输入验证和输出编码      │
│ 5. 监控审计层 - 日志记录和异常检测      │
└────────────────────────────────────────┘
```

**核心原则**：
- **最小权限原则**：仅授予必要的最小权限
- **默认安全**：默认采用最安全的配置
- **纵深防御**：多层安全控制
- **安全失败**：失败时采用安全的方式
- **隐私设计**：从设计阶段就考虑隐私保护

#### 11.1.2 安全威胁模型

| 威胁类别 | 具体威胁 | 风险等级 | 缓解措施 |
|---------|---------|---------|---------|
| **数据泄露** | 未授权访问本地数据 | 高 | 字段加密、设备锁定 |
| **数据泄露** | 网络传输被截获 | 中 | HTTPS强制、证书校验 |
| **注入攻击** | XSS跨站脚本 | 中 | 输出编码、CSP策略 |
| **注入攻击** | SQL注入（云端） | 低 | 参数化查询 |
| **身份冒用** | 会话劫持 | 中 | 安全会话管理 |
| **拒绝服务** | 资源耗尽攻击 | 低 | 速率限制、资源配额 |
| **恶意代码** | 第三方库漏洞 | 中 | 依赖审查、定期更新 |
| **社会工程** | 钓鱼攻击 | 中 | 用户教育、验证提示 |

```javascript
/**
 * 导出前的安全检查
 * @param {Object} options - 导出选项
 */
async function preExportSecurityCheck(options) {
  const { password, exportRange } = options;
  
  // 1. 检查是否包含敏感数据
  const hasSensitiveData = await checkSensitiveDataInExport(exportRange);
  
  // 2. 如果包含敏感数据但未设置密码，警告用户
  if (hasSensitiveData && !password) {
    const userConfirm = await showSecurityWarning(
      '导出的数据包含敏感信息（联系方式、私密备注等），强烈建议设置密码保护。确定要继续吗？'
    );
    
    if (!userConfirm) {
      throw new Error('用户取消导出');
    }
  }
  
  // 3. 密码强度检查
  if (password && !validatePasswordStrength(password)) {
    throw new Error('密码强度不足，请使用至少8位包含字母和数字的密码');
  }
  
  return true;
}

/**
 * 验证密码强度
 * @param {String} password - 密码
 * @returns {Boolean} 是否满足强度要求
 */
function validatePasswordStrength(password) {
  if (password.length < 8) return false;
  
  // 至少包含字母和数字
  const hasLetter = /[a-zA-Z]/.test(password);
  const hasNumber = /[0-9]/.test(password);
  
  return hasLetter && hasNumber;
}
```

```javascript
export function secureDeleteString(str) {
  if (typeof str !== 'string') return;

  str = null;
  
  // 建议垃圾回收

/**
 * 安全删除对象
 * @param {Object} obj - 要删除的对象
 */
export function secureDeleteObject(obj) {
  if (!obj || typeof obj !== 'object') return;
  
  // 递归清除所有属性
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      if (typeof obj[key] === 'object') {
        secureDeleteObject(obj[key]);
      }
      delete obj[key];
    }
  }
}

/**
 * 从数据库彻底删除数据
 * @param {String} tableName - 表名
 * @param {Number} id - 记录ID
 */
export async function permanentDelete(tableName, id) {
  try {
    // 1. 读取数据（用于审计日志）
    const record = await storage[tableName].get(id);
    
    // 2. 记录删除审计日志
    await auditLog.create({
      action: 'DELETE',
      table: tableName,
      recordId: id,
      timestamp: new Date().toISOString()
    });
    
    // 3. 从数据库删除
    await storage[tableName].delete(id);
    
    // 4. 清除内存中的引用
    secureDeleteObject(record);
    
    return { success: true };
  } catch (error) {
    console.error('删除失败:', error);
    throw error;
  }
}
```

```javascript
// utils/request.js

/**
 * 安全的网络请求封装
 * @param {Object} options - 请求选项
 */
export function secureRequest(options) {
  const {
    url,
    method = 'GET',
    data = {},
    header = {}
  } = options;
  
  return new Promise((resolve, reject) => {
    // 1. 验证 URL 是否为 HTTPS
    if (!url.startsWith('https://')) {
      reject(new Error('仅允许 HTTPS 请求'));
      return;
    }
    
    // 2. 添加安全头
    const secureHeaders = {
      ...header,
      'X-Requested-With': 'XMLHttpRequest',
      'X-Client-Version': getAppVersion(),
      'X-Request-ID': generateRequestId()
    };
    
    // 3. 发起请求
    wx.request({
      url,
      method,
      data,
      header: secureHeaders,
      
      // 4. 启用 SSL 证书校验
      enableHttp2: true,
      enableQuic: true,
      
      success(res) {
        // 5. 验证响应
        if (validateResponse(res)) {
          resolve(res.data);
        } else {
          reject(new Error('响应验证失败'));
        }
      },
      
      fail(error) {
        console.error('请求失败:', error);
        reject(error);
      }
    });
  });
}

/**
 * 验证响应安全性
 * @param {Object} response - 响应对象
 */
function validateResponse(response) {
  // 1. 检查状态码
  if (response.statusCode < 200 || response.statusCode >= 300) {
    return false;
  }
  
  // 2. 检查必要的安全头
  const headers = response.header || {};
  
  // 验证内容类型
  const contentType = headers['content-type'] || headers['Content-Type'];
  if (!contentType) {
    console.warn('响应缺少 Content-Type 头');
  }
  
  return true;
}
```

**云函数权限控制**：
```javascript
// cloud/functions/sync/index.js
const cloud = require('wx-server-sdk');
cloud.init();

exports.main = async (event, context) => {
  const { action, data } = event;
  const { OPENID } = cloud.getWXContext();
  
  // 1. 身份验证
  if (!OPENID) {
    return {
      success: false,
      error: 'UNAUTHORIZED',
      message: '未授权访问'
    };
  }
  
  // 2. 操作权限检查
  const hasPermission = await checkPermission(OPENID, action);
  if (!hasPermission) {
    return {
      success: false,
      error: 'FORBIDDEN',
      message: '无权执行此操作'
    };
  }
  
  // 3. 输入验证
  const validationResult = validateInput(action, data);
  if (!validationResult.valid) {
    return {
      success: false,
      error: 'INVALID_INPUT',
      message: validationResult.errors.join(', ')
    };
  }
  
  // 4. 速率限制
  const rateLimitOk = await checkRateLimit(OPENID, action);
  if (!rateLimitOk) {
    return {
      success: false,
      error: 'RATE_LIMIT_EXCEEDED',
      message: '请求过于频繁'
    };
  }
  
  // 5. 执行操作
  try {
    const result = await handleAction(action, data, OPENID);
    
    // 6. 审计日志
    await logAuditEvent({
      userId: OPENID,
      action: action,
      timestamp: new Date(),
      success: true
    });
    
    return {
      success: true,
      data: result
    };
  } catch (error) {
    console.error('操作失败:', error);
    
    // 记录失败审计日志
    await logAuditEvent({
      userId: OPENID,
      action: action,
      timestamp: new Date(),
      success: false,
      error: error.message
    });
    
    return {
      success: false,
      error: 'INTERNAL_ERROR',
      message: '操作失败'
    };
  }
};

/**
 * 检查用户权限
 */
async function checkPermission(openid, action) {
  // 基于角色的访问控制（RBAC）
  const db = cloud.database();
  const user = await db.collection('users').where({ openid }).get();
  
  if (!user.data || user.data.length === 0) {
    return false;
  }
  
  const userRole = user.data[0].role || 'user';
  const permissions = ROLE_PERMISSIONS[userRole] || [];
  
  return permissions.includes(action);
}

/**
 * 速率限制检查
 */
async function checkRateLimit(openid, action) {
  const db = cloud.database();
  const _ = db.command;
  
  // 查询最近1分钟的请求次数
  const oneMinuteAgo = new Date(Date.now() - 60000);
  const recentRequests = await db.collection('request_logs')
    .where({
      userId: openid,
      action: action,
      timestamp: _.gte(oneMinuteAgo)
    })
    .count();
  
  // 每分钟最多30次请求
  return recentRequests.total < 30;
}

// 角色权限配置
const ROLE_PERMISSIONS = {
  admin: ['read', 'write', 'delete', 'sync', 'backup'],
  user: ['read', 'write', 'sync'],
  guest: ['read']
};
```

### 11.4 应用层安全

#### 11.4.1 输入验证和清理

**通用输入验证器**：
```javascript
// utils/validator.js

/**
 * 验证和清理用户输入
 * @param {String} input - 用户输入
 * @param {String} type - 输入类型
 * @returns {Object} { valid: Boolean, cleaned: String, errors: Array }
 */
export function validateAndCleanInput(input, type) {
  const errors = [];
  let cleaned = input;
  
  // 基本验证
  if (cleaned === null || cleaned === undefined) {
    return { valid: false, cleaned: '', errors: ['输入不能为空'] };
  }
  
  // 转换为字符串
  cleaned = String(cleaned);
  
  // 长度限制
  const MAX_LENGTHS = {
    name: 50,
    email: 100,
    phone: 20,
    note: 1000,
    address: 200
  };
  
  if (MAX_LENGTHS[type] && cleaned.length > MAX_LENGTHS[type]) {
    errors.push(`输入长度不能超过 ${MAX_LENGTHS[type]} 个字符`);
    cleaned = cleaned.substring(0, MAX_LENGTHS[type]);
  }
  
  // 类型特定验证
  switch (type) {
    case 'name':
      cleaned = cleanName(cleaned);
      break;
    
    case 'email':
      if (!isValidEmail(cleaned)) {
        errors.push('邮箱格式不正确');
      }
      break;
    
    case 'phone':
      cleaned = cleanPhone(cleaned);
      if (!isValidPhone(cleaned)) {
        errors.push('电话号码格式不正确');
      }
      break;
    
    case 'url':
      if (!isValidURL(cleaned)) {
        errors.push('URL 格式不正确');
      }
      break;
    
    case 'note':
      cleaned = sanitizeHTML(cleaned);
      break;
  }
  
  return {
    valid: errors.length === 0,
    cleaned: cleaned,
    errors: errors
  };
}

/**
 * 清理姓名输入
 */
function cleanName(name) {
  // 移除两端空白
  name = name.trim();
  
  // 移除特殊字符（保留中文、英文、数字、空格、连字符）
  name = name.replace(/[^\u4e00-\u9fa5a-zA-Z0-9\s\-·]/g, '');
  
  // 压缩多个空格为一个
  name = name.replace(/\s+/g, ' ');
  
  return name;
}

/**
 * 清理电话号码
 */
function cleanPhone(phone) {
  // 仅保留数字、加号、连字符、括号
  return phone.replace(/[^\d+\-()]/g, '');
}

/**
 * HTML 清理（防止 XSS）
 */
function sanitizeHTML(html) {
  // 简单的 HTML 转义
  const map = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#x27;',
    '/': '&#x2F;'
  };
  
  return html.replace(/[&<>"'/]/g, char => map[char]);
}

/**
 * 验证邮箱格式
 */
function isValidEmail(email) {
  const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return regex.test(email);
}

/**
 * 验证电话号码格式
 */
function isValidPhone(phone) {
  // 支持中国大陆、香港、台湾、国际号码
  const regex = /^[\d+\-()]{8,20}$/;
  return regex.test(phone);
}

/**
 * 验证 URL 格式
 */
function isValidURL(url) {
  try {
    new URL(url);
    return url.startsWith('http://') || url.startsWith('https://');
  } catch {
    return false;
  }
}
```

**在服务中应用输入验证**：
```javascript
// services/characterService.js
import { validateAndCleanInput } from '../utils/validator';

export async function createCharacter(characterData) {
  // 验证和清理输入
  const validations = {
    displayName: validateAndCleanInput(characterData.displayName, 'name'),
    'nameFields.lastName': validateAndCleanInput(characterData.nameFields?.lastName, 'name'),
    'nameFields.firstName': validateAndCleanInput(characterData.nameFields?.firstName, 'name')
  };
  
  // 检查验证结果
  const errors = [];
  for (const [field, result] of Object.entries(validations)) {
    if (!result.valid) {
      errors.push(`${field}: ${result.errors.join(', ')}`);
    }
  }
  
  if (errors.length > 0) {
    throw new Error(`输入验证失败:\n${errors.join('\n')}`);
  }
  
  // 使用清理后的数据
  const cleanedData = {
    ...characterData,
    displayName: validations.displayName.cleaned,
    nameFields: {
      ...characterData.nameFields,
      lastName: validations['nameFields.lastName'].cleaned,
      firstName: validations['nameFields.firstName'].cleaned
    }
  };
  
  // 继续创建流程...
  return await storage.characters.add(cleanedData);
}
```

```javascript
// utils/safeRender.js

/**
 * 安全地渲染文本（防止 XSS）
 * @param {String} text - 要渲染的文本
 * @returns {String} 安全的文本
 */
export function escapeHTML(text) {
  if (!text) return '';
  
  const map = {
    '&': '&amp;',
    '<': '&lt;',
    '>': '&gt;',
    '"': '&quot;',
    "'": '&#x27;',
    '/': '&#x2F;'
  };
  
  return String(text).replace(/[&<>"'/]/g, char => map[char]);
}

/**
 * 安全地渲染 URL
 * @param {String} url - URL
 * @returns {String} 安全的 URL
 */
export function sanitizeURL(url) {
  if (!url) return '';
  
  // 仅允许 http 和 https 协议
  if (!url.startsWith('http://') && !url.startsWith('https://')) {
    return '';
  }
  
  // 移除 javascript: 等危险协议
  if (url.toLowerCase().includes('javascript:')) {
    return '';
  }
  
  return url;
}
```

### 11.5 身份验证和授权

#### 11.5.1 微信登录安全

```javascript
// services/authService.js

/**
 * 安全的微信登录
 */
export async function secureWechatLogin() {
  try {
    // 1. 获取微信登录凭证
    const loginRes = await wx.login();
    
    if (!loginRes.code) {
      throw new Error('获取登录凭证失败');
    }
    
    // 2. 发送到后端验证
    const authResult = await secureRequest({
      url: 'https://api.guanxi-app.com/auth/wechat',
      method: 'POST',
      data: {
        code: loginRes.code,
        timestamp: Date.now(),
        nonce: generateNonce()
      }
    });
    
    // 3. 验证返回的 token
    if (!authResult.token || !validateToken(authResult.token)) {
      throw new Error('token 验证失败');
    }
    
    // 4. 安全存储 token
    await secureTokenStorage.set(authResult.token);
    
    // 5. 设置 token 过期时间
    const expiresIn = authResult.expiresIn || 7200;  // 默认2小时
    setTimeout(() => {
      this.refreshToken();
    }, (expiresIn - 300) * 1000);  // 提前5分钟刷新
    
    return {
      success: true,
      user: authResult.user
    };
  } catch (error) {
    console.error('登录失败:', error);
    throw error;
  }
}

/**
 * Token 刷新
 */
export async function refreshToken() {
  try {
    const oldToken = await secureTokenStorage.get();
    
    const result = await secureRequest({
      url: 'https://api.guanxi-app.com/auth/refresh',
      method: 'POST',
      header: {
        'Authorization': `Bearer ${oldToken}`
      }
    });
    
    await secureTokenStorage.set(result.token);
    
    return result.token;
  } catch (error) {
    console.error('Token 刷新失败:', error);
    // Token 过期，需要重新登录
    await this.logout();
    throw error;
  }
}

/**
 * 验证 Token 格式
 */
function validateToken(token) {
  if (!token || typeof token !== 'string') {
    return false;
  }
  
  // JWT token 应该有三部分
  const parts = token.split('.');
  if (parts.length !== 3) {
    return false;
  }
  
  return true;
}

/**
 * 生成随机 nonce
 */
function generateNonce() {
  return Math.random().toString(36).substring(2, 15);
}
```

```javascript
// utils/secureTokenStorage.js

class SecureTokenStorage {
  constructor() {
    this.storageKey = '_auth_token';
  }
  
  /**
   * 存储 token（加密）
   */
  async set(token) {
    try {
      // 使用设备密钥加密 token
      const deviceKey = await this.getDeviceKey();
      const encrypted = CryptoJS.AES.encrypt(token, deviceKey).toString();
      
      wx.setStorageSync(this.storageKey, encrypted);
      
      // 在内存中也保留一份（会话期间有效）
      getApp().globalData.authToken = token;
      
      return true;
    } catch (error) {
      console.error('Token 存储失败:', error);
      return false;
    }
  }
  
  /**
   * 获取 token（解密）
   */
  async get() {
    try {
      // 优先从内存获取
      const memoryToken = getApp().globalData.authToken;
      if (memoryToken) {
        return memoryToken;
      }
      
      // 从存储读取并解密
      const encrypted = wx.getStorageSync(this.storageKey);
      if (!encrypted) {
        return null;
      }
      
      const deviceKey = await this.getDeviceKey();
      const decrypted = CryptoJS.AES.decrypt(encrypted, deviceKey);
      const token = decrypted.toString(CryptoJS.enc.Utf8);
      
      // 缓存到内存
      getApp().globalData.authToken = token;
      
      return token;
    } catch (error) {
      console.error('Token 读取失败:', error);
      return null;
    }
  }
  
  /**
   * 清除 token
   */
  async clear() {
    wx.removeStorageSync(this.storageKey);
    if (getApp().globalData) {
      delete getApp().globalData.authToken;
    }
  }
  
  /**
   * 获取设备密钥
   */
  async getDeviceKey() {
    let deviceKey = wx.getStorageSync('_device_key');
    
    if (!deviceKey) {
      deviceKey = CryptoJS.lib.WordArray.random(16).toString(CryptoJS.enc.Hex);
      wx.setStorageSync('_device_key', deviceKey);
    }
    
    return deviceKey;
  }
}

export default new SecureTokenStorage();
```

```javascript
// services/auditLog.js

class AuditLogger {
  constructor() {
    this.storeName = 'audit_logs';
  }
  
  /**
   * 记录审计事件
   * @param {Object} event - 审计事件
   */
  async log(event) {
    const auditRecord = {
      id: Date.now(),
      timestamp: new Date().toISOString(),
      userId: await this.getCurrentUserId(),
      action: event.action,
      resource: event.resource,
      resourceId: event.resourceId,
      result: event.result || 'success',
      details: event.details || {},
      ip: await this.getClientIP(),
      userAgent: this.getUserAgent()
    };
    
    try {
      await storage.audit_logs.add(auditRecord);
      
      // 如果是敏感操作，额外记录到云端
      if (this.isSensitiveAction(event.action)) {
        await this.logToCloud(auditRecord);
      }
    } catch (error) {
      console.error('审计日志记录失败:', error);
    }
  }
  
  /**
   * 查询审计日志
   */
  async query(filter = {}) {
    const { userId, action, startTime, endTime, limit = 100 } = filter;
    
    let query = storage.audit_logs.toCollection();
    
    if (userId) {
      query = query.filter(log => log.userId === userId);
    }
    
    if (action) {
      query = query.filter(log => log.action === action);
    }
    
    if (startTime) {
      query = query.filter(log => log.timestamp >= startTime);
    }
    
    if (endTime) {
      query = query.filter(log => log.timestamp <= endTime);
    }
    
    const logs = await query.limit(limit).reverse().toArray();
    
    return logs;
  }
  
  /**
   * 判断是否为敏感操作
   */
  isSensitiveAction(action) {
    const sensitiveActions = [
      'DELETE_CHARACTER',
      'DELETE_GUANXI',
      'EXPORT_DATA',
      'IMPORT_DATA',
      'CHANGE_PASSWORD',
      'ENABLE_ENCRYPTION'
    ];
    
    return sensitiveActions.includes(action);
  }
  
  /**
   * 记录到云端（可选）
   */
  async logToCloud(record) {
    if (!this.cloudLoggingEnabled()) {
      return;
    }
    
    try {
      await wx.cloud.callFunction({
        name: 'auditLog',
        data: {
          action: 'create',
          record: record
        }
      });
    } catch (error) {
      console.error('云端审计日志记录失败:', error);
    }
  }
  
  async getCurrentUserId() {
    // 获取当前用户ID
    return getApp().globalData.userId || 'anonymous';
  }
  
  async getClientIP() {
    // 微信小程序无法直接获取IP，返回空
    return '';
  }
  
  getUserAgent() {
    const systemInfo = wx.getSystemInfoSync();
    return `${systemInfo.platform} ${systemInfo.system} WeChat/${systemInfo.version}`;
  }
  
  cloudLoggingEnabled() {
    return wx.getStorageSync('cloud_logging_enabled') === true;
  }
}

export default new AuditLogger();
```

```javascript
// utils/errorMonitor.js

class ErrorMonitor {
  constructor() {
    this.errorQueue = [];
    this.maxQueueSize = 50;
  }
  
  /**
   * 初始化错误监控
   */
  init() {
    // 监听未捕获的错误
    wx.onError((error) => {
      this.captureError({
        type: 'uncaught',
        message: error,
        timestamp: new Date().toISOString()
      });
    });
    
    // 监听未处理的 Promise 拒绝
    wx.onUnhandledRejection((res) => {
      this.captureError({
        type: 'unhandled_rejection',
        message: res.reason,
        timestamp: new Date().toISOString()
      });
    });
  }
  
  /**
   * 捕获错误
   */
  captureError(error) {
    console.error('捕获到错误:', error);
    
    // 添加到队列
    this.errorQueue.push(error);
    
    // 保持队列大小
    if (this.errorQueue.length > this.maxQueueSize) {
      this.errorQueue.shift();
    }
    
    // 保存到本地存储
    this.saveErrorLog(error);
    
    // 如果是严重错误，立即上报
    if (this.isCriticalError(error)) {
      this.reportError(error);
    }
  }
  
  /**
   * 判断是否为严重错误
   */
  isCriticalError(error) {
    const criticalPatterns = [
      /database/i,
      /encryption/i,
      /authentication/i,
      /authorization/i
    ];
    
    return criticalPatterns.some(pattern => 
      pattern.test(error.message || error.type)
    );
  }
  
  /**
   * 保存错误日志到本地
   */
  async saveErrorLog(error) {
    try {
      const errorLog = {
        ...error,
        id: Date.now(),
        systemInfo: wx.getSystemInfoSync(),
        appVersion: getApp().globalData.version
      };
      
      await storage.error_logs.add(errorLog);
    } catch (e) {
      console.error('保存错误日志失败:', e);
    }
  }
  
  /**
   * 上报错误（可选）
   */
  async reportError(error) {
    if (!this.reportingEnabled()) {
      return;
    }
    
    try {
      await secureRequest({
        url: 'https://api.guanxi-app.com/errors/report',
        method: 'POST',
        data: {
          error: error,
          systemInfo: wx.getSystemInfoSync(),
          timestamp: new Date().toISOString()
        }
      });
    } catch (e) {
      console.error('错误上报失败:', e);
    }
  }
  
  reportingEnabled() {
    return wx.getStorageSync('error_reporting_enabled') === true;
  }
  
  /**
   * 获取错误统计
   */
  getStatistics() {
    return {
      total: this.errorQueue.length,
      byType: this.groupByType(),
      recent: this.errorQueue.slice(-10)
    };
  }
  
  groupByType() {
    const grouped = {};
    
    this.errorQueue.forEach(error => {
      const type = error.type || 'unknown';
      grouped[type] = (grouped[type] || 0) + 1;
    });
    
    return grouped;
  }
}

export default new ErrorMonitor();
```
