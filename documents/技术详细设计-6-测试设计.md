## 6. 测试设计

### 6.1 单元测试

**测试范围**：
- 工具函数
- 关系类型类
- 验证逻辑
- 算法函数

**测试框架**：Jest

**示例**：
```javascript
// tests/types/relative.type.test.js
const RelativeType = require('../../types/relative.type');

describe('RelativeType', () => {
  let relativeType;

  beforeEach(() => {
    relativeType = new RelativeType();
  });

  test('should have correct config', () => {
    expect(relativeType.id).toBe('relative');
    expect(relativeType.name).toBe('亲戚关系');
    expect(relativeType.category).toBe('family');
  });

  test('should validate required fields', () => {
    const result = relativeType.validate({});
    expect(result.valid).toBe(false);
    expect(result.errors).toContain('称谓不能为空');
  });

  test('should validate valid attributes', () => {
    const result = relativeType.validate({
      title: '父亲',
      lineage: '直系',
      generation: '长辈'
    });
    expect(result.valid).toBe(true);
    expect(result.errors).toHaveLength(0);
  });
});
```
